# plot 6 samples of each individual
samples_number = 6
def plot_samples(images, titles, h, w, n_row, n_col):
    plt.figure(figsize=(n_col, 2 * n_row))
    for i in range(samples_number * n_classes):
        plt.subplot(n_row, n_col, i + 1)
        plt.imshow(images[i].reshape((h, w)))
        plt.title(titles[i], size=6)
        plt.xticks(())
        plt.yticks(())

# get unique images and reshape them to plot
single_faces = [X_train[y_train == lab][:samples_number] for lab in np.unique(y_train)]
single_faces = np.vstack(single_faces).reshape((samples_number * n_classes, h, w))

# include title names for each person
titles = [n for name in target_names for n in [name] * samples_number]

plot_samples(single_faces, titles, h, w, n_row=n_classes, n_col=samples_number)

def transform_input_data(X_train, X_test):
  """
  Preprocess the input data to work for your model.
  """
  from sklearn import preprocessing
  scaler = preprocessing.MinMaxScaler()
  X_train_transformed = scaler.fit_transform(X_train).reshape(-1, 1, h, w)
  X_test_transformed = scaler.transform(X_test).reshape(-1, 1, h, w)

  return X_train_transformed, X_test_transformed

# CNN to find patterns on the images
# Input -> Conv -> Max Pooling -> Conv 2 -> Max Pooling -> Flattening -> Output

class NeuralNetwork(nn.Module):
    def __init__(self):
        super(NeuralNetwork, self).__init__()
        self.conv1 = nn.Conv2d(1, 32, (3, 3))
        self.conv2 = nn.Conv2d(32, 64, (3, 3))
        self.conv2_drop = nn.Dropout2d(p=0.5)
        self.fc1 = nn.Linear(64 * 11 * 7, 90)
        self.fc2 = nn.Linear(90, 7)
        self.fc1_drop = nn.Dropout(p=0.5)

    def forward(self, x):
        x = self.conv1(x)
        x = F.max_pool2d(x, 2)
        x = F.relu(x)

        x = self.conv2(x)
        x = self.conv2_drop(x)
        x = F.max_pool2d(x, 2)
        x = F.relu(x)

        x = x.view(-1, 64 * 11 * 7)
        x = self.fc1_drop(self.fc1(x))
        x = F.relu(x)
        x = torch.softmax(self.fc2(x), dim=-1)
        return x

from torchsummary import summary

# Use learning rate 0.001, Adam Optimizer and run for 100 epochs

model = NeuralNetClassifier(
        NeuralNetwork,
        max_epochs=120,
        lr=1e-3,
        optimizer=torch.optim.Adam,
        device=device,
        train_split=skorch.dataset.ValidSplit(cv=5, stratified=True),
        verbose=0)

# transform the input data to fit in the model
X_train_transformed, X_test_transformed = transform_input_data(X_train, X_test)

model.fit(X_train_transformed, y_train)

y_pred = model.predict(X_test_transformed)
f1 = f1_score(y_test, y_pred, average="weighted")
